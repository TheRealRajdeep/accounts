# `eth_sendTransactionSync`

Signs, broadcasts, and waits for a transaction receipt. Returns the full receipt instead of just the hash.

## Request

```ts
type Request = {
  method: 'eth_sendTransactionSync'
  params: [TransactionRequest]
}

type TransactionRequest = {
  /** Access list entries. */
  accessList?: { address: `0x${string}`; storageKeys: `0x${string}`[] }[]
  /** Batch of calls. */
  calls?: { data?: `0x${string}`; to?: `0x${string}`; value?: `0x${string}` }[]
  /** Chain ID (hex-encoded number). */
  chainId?: `0x${string}`
  /** Top-level calldata. */
  data?: `0x${string}`
  /**
   * Fee payer override. `true` enables the relay default, `false` disables
   * it, or a URL string points at a custom fee payer service.
   */
  feePayer?: boolean | string
  /** Pre-computed fee payer signature payload. */
  feePayerSignature?: Record<string, unknown>
  /** Fee token to use. If omitted, the relay picks the user's best token. */
  feeToken?: `0x${string}`
  /** Sender address. */
  from?: `0x${string}`
  /** Gas limit (hex-encoded bigint). */
  gas?: `0x${string}`
  /** Key authorization for access key transactions. */
  keyAuthorization?: KeyAuthorization
  /** Raw key material. */
  keyData?: `0x${string}`
  /** Address of the key being used to sign. */
  keyId?: `0x${string}`
  /** Key type used to sign. */
  keyType?: 'secp256k1' | 'p256' | 'webAuthn'
  /** Max fee per gas (hex-encoded bigint). */
  maxFeePerGas?: `0x${string}`
  /** Max priority fee per gas (hex-encoded bigint). */
  maxPriorityFeePerGas?: `0x${string}`
  /** Nonce (hex-encoded number). */
  nonce?: `0x${string}`
  /** Nonce key (hex-encoded bigint). */
  nonceKey?: `0x${string}`
  /** Top-level recipient. */
  to?: `0x${string}`
  /** Value to transfer (hex-encoded bigint). */
  value?: `0x${string}`
  /** Valid after timestamp (hex-encoded number, unix seconds). */
  validAfter?: `0x${string}`
  /** Valid before timestamp (hex-encoded number, unix seconds). */
  validBefore?: `0x${string}`
}
```

## Response

```ts
type Response = TransactionReceipt

type TransactionReceipt = {
  /** Blob base fee per gas (post-EIP-4844). */
  blobGasPrice?: bigint
  /** Total blob gas used by the transaction. */
  blobGasUsed?: bigint
  /** Block hash containing the transaction. */
  blockHash: `0x${string}`
  /** Block number containing the transaction. */
  blockNumber: bigint
  /** Deployed contract address (when this is a contract creation). */
  contractAddress: `0x${string}` | null
  /** Cumulative gas used in the block up to and including this transaction. */
  cumulativeGasUsed: bigint
  /** Effective gas price paid. */
  effectiveGasPrice: bigint
  /** Address that paid the fees on the user's behalf, if any. */
  feePayer?: `0x${string}`
  /** Token used to pay fees, if any. */
  feeToken?: `0x${string}`
  /** Sender address. */
  from: `0x${string}`
  /** Gas used by this transaction. */
  gasUsed: bigint
  /** Logs emitted by the transaction. */
  logs: {
    address: `0x${string}`
    blockHash: `0x${string}`
    blockNumber: bigint
    data: `0x${string}`
    logIndex: number
    removed: boolean
    topics: readonly `0x${string}`[]
    transactionHash: `0x${string}`
    transactionIndex: number
  }[]
  /** Bloom filter for the receipt logs. */
  logsBloom: `0x${string}`
  /** State root after the transaction (pre-Byzantium receipts). */
  root?: `0x${string}`
  /** Transaction status. */
  status: 'success' | 'reverted'
  /** Recipient address. */
  to: `0x${string}` | null
  /** Transaction hash. */
  transactionHash: `0x${string}`
  /** Index of the transaction within the block. */
  transactionIndex: number
  /** EIP-2718 transaction type (hex-encoded). */
  type: `0x${string}`
}
```

## Example

```ts
const receipt = await provider.request({
  method: 'eth_sendTransactionSync',
  params: [{
    to: '0x...',
    value: '0x0',
    data: '0x...',
  }],
})
```
